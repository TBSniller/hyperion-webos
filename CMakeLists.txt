cmake_minimum_required(VERSION 3.6)
project(hyperion-webos VERSION 1.0 LANGUAGES C)

add_subdirectory(tv-native-apis)
add_subdirectory(flatccrt)

find_package(PkgConfig REQUIRED)

pkg_check_modules(OpenGL REQUIRED IMPORTED_TARGET egl glesv2)

add_executable(hyperion-webos
    src/capture_newtv.c
    src/renderer.c
    src/hyperion_client.c
 )
target_include_directories(hyperion-webos PRIVATE fbs incbin)
target_compile_definitions(hyperion-webos PRIVATE SOURCE_DIR="${CMAKE_CURRENT_SOURCE_DIR}/src")
target_link_libraries(hyperion-webos vt vtcapture gm flatccrt pthread PkgConfig::OpenGL)
